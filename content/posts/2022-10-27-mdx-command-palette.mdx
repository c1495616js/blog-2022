```ts fileName=index.tsx
import React, { forwardRef, useMemo } from 'react';
import {
  ActionId,
  ActionImpl,
  KBarAnimator,
  KBarPortal,
  KBarPositioner,
  KBarProvider,
  KBarResults,
  KBarSearch,
  Priority,
  useMatches,
} from 'kbar';
import { Box, Typography, colors } from '@mui/material';
import { styled } from '@mui/system';

type Props = {
  children: React.ReactNode;
};

interface ResultItemProps {
  action: ActionImpl;
  active: boolean;
  currentRootActionId: ActionId;
}
type Ref = HTMLDivElement;

const ResultItem = forwardRef<Ref, ResultItemProps>(
  (
    {
      action,
      active,
      currentRootActionId,
    }: {
      action: ActionImpl;
      active: boolean;
      currentRootActionId: ActionId;
    },
    ref: React.Ref<HTMLDivElement>
  ) => {
    const ancestors = useMemo(() => {
      if (!currentRootActionId) return action.ancestors;
      const index = action.ancestors.findIndex(
        (ancestor) => ancestor.id === currentRootActionId
      );
      return action.ancestors.slice(index + 1);
    }, [action.ancestors, currentRootActionId]);

    const StyledKbd = styled('kbd')`
      display: flex;
      padding-top: 0.5rem;
      padding-bottom: 0.5rem;
      padding-left: 0.75rem;
      padding-right: 0.75rem;
      justify-content: space-between;
      align-items: center;
      border-radius: 0.375rem;
      cursor: pointer;
      background-color: #fff;
    `;

    return (
      <Box
        ref={ref}
        sx={{
          display: 'flex',
          flexDirection: 'row',
          justifyContent: 'justify-between',
          alignItems: 'center',
          cursor: 'pointer',
          height: '48px',
          padding: '0.5rem 1.5rem',
          color: (theme) => theme.palette.text.primary,
          ...(active
            ? {
                backgroundColor: '#F1F0F1',
                borderLeft: (theme) =>
                  `2.5px solid ${theme.palette.text.primary}`,
              }
            : {}),
        }}
      >
        <Box sx={{ display: 'flex', alignItems: 'center', gap: 2 }}>
          {action.icon && action.icon}
          <Box sx={{ display: 'flex', flexDirection: 'column' }}>
            <Box>
              {ancestors.length > 0 &&
                ancestors.map((ancestor) => (
                  <React.Fragment key={ancestor.id}>
                    <Typography variant="body1" sx={{ marginRight: 3 }}>
                      {ancestor.name}
                    </Typography>
                    <Typography variant="body1" sx={{ marginRight: 3 }}>
                      â€º
                    </Typography>
                  </React.Fragment>
                ))}
              <Typography variant="caption" sx={{}}>
                {action.name}
              </Typography>
            </Box>
            {action.subtitle && (
              <span className="text-sm">{action.subtitle}</span>
            )}
          </Box>
        </Box>
        {action.shortcut?.length ? (
          <Box
            sx={{ display: 'grid', gap: '0.5rem', gridAutoFlow: 'column' }}
            aria-hidden
            className="grid grid-flow-col gap-2"
          >
            {action.shortcut.map((sc) => (
              <StyledKbd key={sc}>{sc}</StyledKbd>
            ))}
          </Box>
        ) : null}
      </Box>
    );
  }
);

function RenderResults() {
  const { results, rootActionId } = useMatches();

  return (
    <KBarResults
      items={results}
      onRender={({ item, active }) =>
        typeof item === 'string' ? (
          <Box sx={{ pl: 3, pt: 2, pb: 1, color: colors.grey[600] }}>
            {item}
          </Box>
        ) : (
          <ResultItem
            action={item}
            active={active}
            currentRootActionId={rootActionId || ''}
          />
        )
      }
    />
  );
}

const StyledKBarPositioner = styled(KBarPositioner)`
  display: "flex";
  alignItems: "center;
  z-index: 20;
  background-color: rgba(156, 163, 175, 0.7);
  backdrop-filter: blur(4px);
`;

const StyledKBarAnimator = styled(KBarAnimator)`
  box-sizing: content-box;
  overflow-x: hidden;
  overflow-y: auto;
  padding: 1rem 0;
  width: 50%;
  border-radius: 0.75rem;
  border-width: 1px;
  border-color: #9ca3af;
  background-color: #fff;
  box-shadow: 0 1px 3px 0 rgb(0 0 0 / 0.1), 0 1px 2px -1px rgb(0 0 0 / 0.1);
  z-index: 30;
`;

const StyledKBarSearch = styled(KBarSearch)`
  display: flex;
  padding: 0;
  background-color: transparent;
  width: 90%;
  height: 2rem;
  outline: 0;
  margin: 0 auto;
  border: none;
`;

function CommandBar() {
  return (
    <KBarPortal>
      <StyledKBarPositioner>
        <StyledKBarAnimator>
          <StyledKBarSearch />
          <RenderResults />
        </StyledKBarAnimator>
      </StyledKBarPositioner>
    </KBarPortal>
  );
}

export default function CommandPalette({ children }: Props) {
  // const { isLoading, attachments, attachmentBeingViewed, previewAttachment } = useWelcomeGuideFile();
  const {
    actions: { setOccupation },
  } = useWelcomeGuideStore();
  const navigate = useNavigate();
  /**
   * Can extend more actions here in the future
   */
  const actions = [
    {
      id: 'xxxxx',
      name: 'action',
      keywords: 'yyyyyy',
      perform: () => {},
      icon: <IconComponent />,
      section: {
        name: 'zzzz',
        priority: Priority.HIGH,
      },
    },
  ];

  return (
    <KBarProvider actions={actions}>
      <CommandBar />
      {children}
    </KBarProvider>
  );
}
```

```ts fileName=
import React from 'react';
import { useKBar } from 'kbar';
import { Box, Tooltip, Button } from '@mui/material';
import KeyboardCommandKeyIcon from '@mui/icons-material/KeyboardCommandKey';

export default function CommandPaletteButton() {
  const { query } = useKBar();
  return (
    <>
      <Tooltip title="Command Palette">
        <Box sx={{ ml: 1.5 }}>
          <Button onClick={query?.toggle}>
            <KeyboardCommandKeyIcon sx={{ height: 24, width: 24 }} />
          </Button>
        </Box>
      </Tooltip>
    </>
  );
}
```
